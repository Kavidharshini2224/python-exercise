n=int(input("Enter the Digit from 0 to 9: "))
print("Entered Digit is : ",end='')
if n==0:
    print("zero")
elif n==1:
    print("one")
elif n==2:
    print("Two")
elif n==3:
    print("Three")
elif n==4:
    print("Four")
elif n==5:
    print("Five")
else:
    print("Not a digit")
